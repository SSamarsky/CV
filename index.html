function rgb(r, g, b) {
    let hexColor = '';
    function transformNum(num) {
        let str = '';
        if (num > 255) {
            num = 255;
        } else if (num < 0) {
            num = 0;
        }
        if (num < 16) {
            str = str + 0 + num.toString(16);
            return str;
        } else {
            str = str + num.toString(16);
            return str;
        }
    }
    hexColor = hexColor + transformNum(r) + transformNum(g) + transformNum(b);
    return hexColor.toUpperCase();
}
